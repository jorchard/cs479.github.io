Make a Population¶

In [3]:
r = 0.2
pop = lif.MakePopulation(100, [-r,r], [80, 150])

# Get decoders
P = 500
X = np.linspace(-r,r,P, endpoint=True)
A = lif.Stim2Rate(X, pop)
D = lif.Decoders(A, X)  # Identity decoders

# Recurrent dynamics
E = lif.Encoders(pop)
W = np.dot(D, E);

# Make an interesting input function.
P = 1000
dt = 0.001
t2 = np.array( range(P) ) * dt

X = np.zeros([P,1])
X[1:200,0] = 0.5;
X[300:400,0] = -0.8;
X[600:700,0] = 0.4;

plt.plot(t2, X);
