Set up Population¶

In [3]:
N = 100
lifA = lif.GenerateLIFPopulation(N, [-1, 1], [80, 200], dim=1)

P = 1000
X = np.random.rand(P)*2-1

# Compute optimal linear decoding weights, for comparison
A = lif.Stim2Rate(X, lifA)
DA_true = lif.Decoders(A, X).squeeze()

# Initialize with random decoding weights
DA_rand = np.random.normal(size=N)*np.max(np.abs(DA_true))
DA = copy.deepcopy(DA_rand)

Learning Iteration¶

In [5]:
iters = 100
batchsize = 50
kappa = 0.1 / 200 / N / batchsize
In [6]:
for p in range(iters):
    # Choose some samples
    X = np.random.rand(batchsize)*2-1
    
    # Get neural activities
    A = lif.Stim2Rate(X, lifA)
    
    # Update formula
    err = A@DA - X
    delta_DA = kappa * np.dot(A.T,  err)
    DA -= delta_DA
