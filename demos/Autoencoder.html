class BinaryWorld(torch.utils.data.Dataset):
    def __init__(self, A, noise=0.):
        super().__init__()
        self.x = A.clone() + torch.normal(torch.zeros_like(A))*noise
        self.y = A.clone()
        
    def __len__(self):
        return len(self.x)
    
    def __getitem__(self, idx):
        return self.x[idx], self.y[idx]
    
    def inputs(self):
        return self.x
    
    def targets(self):
        return self.y

class AE(nn.Module):
    def __init__(self, input_dim=8, latent_dim=3):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, latent_dim+2),
            nn.Tanh(),
            nn.Linear(latent_dim+2, latent_dim),
            nn.Sigmoid(),
            )
        
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, latent_dim+2),
            nn.Tanh(),
            nn.Linear(latent_dim+2, input_dim),
            nn.Sigmoid(),
            )
        
    def forward(self, x):
        self.h = self.encoder(x)
        return self.decoder(self.h)

# Useful functions
def IsScalar(x):
    if type(x) in (list, np.ndarray,):
        return False
    else:
        return True

def Thresh(x):
    if IsScalar(x):
        val = 1 if x>0 else -1
    else:
        val = np.ones_like(x)
        val[x<0] = 0.
    return val

def Hamming(x, y):
    '''
        d = Hamming(x,y)
        
        Hamming distance between two binary vectors x and y.
        It's the number of digits that differ.
        
        Inputs:
          x and y are arrays of binary vectors, and can be either {0,1} or {-1,1}
        
        Output:
          d is the number of places where the inputs differ
    '''
    d = []
    for xx, yy in zip(x,y):
        dd = 0.
        for xxx,yyy in zip(xx,yy):
            if xxx==1 and yyy!=1:
                dd += 1.
            elif yyy==1 and xxx!=1:
                dd += 1.
        d.append(dd)
    return d

def Perturb(x, p=0.1):
    '''
        y = Perturb(x, p=0.1)
        
        Apply binary noise to x. With probability p, each bit will be randomly
        set to -1 or 1.
        
        Inputs:
          x is an array of binary vectors of {-1,1}
          p is the probability of each bit being randomly flipped
        
        Output:
          y is an array of binary vectors of {-1,1}
    '''
    y = deepcopy(x)
    for yy in y:
        for k in range(len(yy)):
            if np.random.rand()<p:
                yy[k] = Thresh(np.random.randint(2)*2-1)
    return y

class Letters(torch.utils.data.Dataset):
    def __init__(self, noise=0.):
        super().__init__()
        letters = []
        letters.append(imread('images/A.png'))
        letters.append(imread('images/B.png'))
        letters.append(imread('images/C.png'))
        letters.append(imread('images/E.png'))
        self.n = len(letters)
        N = len(letters[0].flatten())
        self.x = np.zeros((self.n, N))
        for idx,img in enumerate(letters):
            self.x[idx,:] = Thresh(np.array([img.flatten()-0.5]))
        self.x = torch.tensor(self.x, dtype=torch.float)
        self.y = self.x.clone()
        self.x += torch.normal(torch.zeros_like(self.x))*noise
        
    def __len__(self):
        return len(self.x)
    
    def __getitem__(self, idx):
        return self.x[idx], self.y[idx]
    
    def inputs(self):
        return self.x
    
    def targets(self):
        return self.y
    
    def plot(self, x=None):
        plt.figure(figsize=(16,4))
        for k in range(self.n):
            plt.subplot(1,self.n,k+1);
            cm = 'tab20c_r'
            cm = 'jet'
            if x is None:
                plt.imshow(np.reshape(self.x[k].detach().numpy(), (6,6)), cmap=cm);
                plt.axis('off');
            else:
                plt.imshow(np.reshape(x[k].detach().numpy(), (6,6)), cmap=cm);
                plt.axis('off');

class LetterAE(nn.Module):
    def __init__(self, input_dim=8, latent_dim=3):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, 10),
            nn.ReLU(),
            nn.Linear(10, latent_dim),
            nn.Sigmoid()
            )
        
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, 10),
            nn.ReLU(),
            nn.Linear(10, input_dim),
            nn.Sigmoid()
            )
        
    def forward(self, x):
        self.h = self.encoder(x)
        return self.decoder(self.h)
