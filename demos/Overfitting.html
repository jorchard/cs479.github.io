class LinearDS(torch.utils.data.Dataset):
    def __init__(self, n=100, noise=0.1, seed=None):
        def myfunc(x):
            return 0.4*x+0.9
        if seed is not None:
            torch.manual_seed(seed)
        self.x = torch.linspace(-1, 1, n) + torch.normal(mean=torch.zeros((n)))*noise/5.   # x in [-1,1]
        self.y = myfunc(self.x) + torch.normal(mean=torch.zeros((n)))*noise
        perm = torch.randperm(n)
        self.x = self.x[perm]
        self.y = self.y[perm]
        self.xlabel = 'Dose'
        self.ylabel = 'Blood Sugar'

    def __len__(self):
        return len(self.x)
    
    def __getitem__(self, idx):
        return self.x[idx], self.y[idx]
    
    def inputs(self):
        return self.x.reshape( (len(self.x),1) )
    
    def targets(self):
        return self.y.reshape( (len(self.y),1) )
    
    def plot(self, *args, **kwargs): 
        plt.plot(self.x, self.y, *args, **kwargs)
        plt.xlabel('Dose', fontsize=14); plt.ylabel('Blood Sugar', fontsize=14);

Train a Network¶

class Regr(torch.nn.Module):
    def __init__(self, nodes=100):
        super().__init__()
        self.lyrs = torch.nn.ModuleList()
        self.lyrs.append(torch.nn.Linear(1, nodes))
        self.lyrs.append(torch.nn.ReLU())
        self.lyrs.append(torch.nn.Linear(nodes, nodes))
        self.lyrs.append(torch.nn.ReLU())
        self.lyrs.append(torch.nn.Linear(nodes, 1))
        self.loss_fcn = torch.nn.MSELoss(reduction='mean')
    
    def forward(self, x):
        y = x
        for lyr in self.lyrs:
            y = lyr(y)
        return y
    
    def learn(self, x, t, epochs=100, lr=0.1, test=None):
        losses = []
        test_losses = []
        for epoch in range(epochs):
            y = self(x)
            loss = self.loss_fcn(y, t)
            if test is not None:
                loss_test = self.loss_fcn(self(test.inputs()), test.targets())
                test_losses.append( loss_test.item() )
            losses.append(loss.item())
            self.zero_grad()
            loss.backward()
            with torch.no_grad():
                for p in self.parameters():
                    p -= lr*p.grad
        losses = np.array(losses)
        plt.plot(losses);
        if test is not None:
            test_losses = np.array(test_losses)
            plt.plot(test_losses, 'r')
            plt.legend(['Train', 'Test'])
        plt.yscale('log');
        plt.xlabel('Epoch', fontsize=12); plt.ylabel('log Loss', fontsize=12);
        print(f'Final loss = {loss}')
        if test is None:
            return losses
        else:
            return losses, test_losses
