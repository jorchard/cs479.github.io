xmin = -1
xmax = 1

N = 50  # Try 40 and 200
P = 1000

pop = lif.MakePopulation(N, [xmin, xmax], [80, 150])


# Behaviour Sampling
X1 = np.array( np.linspace(xmin, xmax, P, endpoint=True) )
X = X1
A = lif.Stim2Rate(X1, pop)

# Identity Decoders
D = lif.Decoders(A, X1)

# Find fixed points

x_error = np.dot(A,D) - X1[:,np.newaxis]

stable = []
unstable = []

for idx in range(2,len(X1)):
    if x_error[idx-1]<0 and x_error[idx]>0:
        unstable.append(X1[idx])
    elif x_error[idx-1]>0 and x_error[idx]<0:
        stable.append(X1[idx])

plt.plot(X1, np.dot(A,D)-X1[:,np.newaxis])
plt.plot(stable, np.zeros(np.shape(stable)), 'ro')
plt.plot(unstable, np.zeros(np.shape(unstable)), 'kx')
plt.plot([X1[0], X1[-1]], [0, 0], 'k--');
plt.xlabel(r'Input $x$'); plt.ylabel(r'Error $(\hat{x}-x)$');
