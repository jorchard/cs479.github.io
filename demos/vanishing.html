# Make a PyTorch wrapper for my own Annuli dataset
class Annuli(torch.utils.data.Dataset):
    def __init__(self, n=300):
        super().__init__()
        np_ds = utils.Annuli(n=n)  # heavy lifting done by NumPy code
        self.x = torch.tensor(np_ds.inputs(), dtype=torch.float32)
        self.t = torch.argmax(torch.tensor(np_ds.targets()), axis=1, keepdim=False)
        
    def __len__(self):
        return len(self.x)
    
    def __getitem__(self, idx):
        return self.x[idx], self.t[idx]
    
    def inputs(self):
        return self.x
    
    def targets(self):
        return self.t
    
    def plot(self, labels=None, *args, **kwargs): 
        if labels is None:
            labels = self.t
        colour_options = ['y', 'r', 'g', 'b', 'k']
        if len(labels.size())>1 and len(labels[0])>1:
            # one-hot labels
            cidx = torch.argmax(labels, axis=1)
        else:
            # binary labels
            cidx = labels
        colours = [colour_options[k] for k in cidx]
        plt.scatter(self.x[:,0].detach(), self.x[:,1].detach(), color=colours, marker='.')
        plt.axis('equal');

class Shallow(nn.Module):
    def __init__(self):
        super().__init__()
        self.lyr = nn.ModuleList()
        self.lyr.append(nn.Linear(2,100),)
        self.lyr.append(nn.Tanh())
        self.lyr.append(nn.Linear(100,3)); self.lyr.append(nn.LogSoftmax(dim=1))
        
    def forward(self, x):
        y = x
        for l in self.lyr:
            y = l(y)
        return y
    
class Deep(nn.Module):
    def __init__(self):
        super().__init__()
        self.lyr = nn.ModuleList()
        self.lyr.append(nn.Linear(2,25)); self.lyr.append(nn.Tanh())
        self.lyr.append(nn.Linear(25,25)); self.lyr.append(nn.Tanh())
        self.lyr.append(nn.Linear(25,25)); self.lyr.append(nn.Tanh())
        self.lyr.append(nn.Linear(25,25)); self.lyr.append(nn.Tanh())
        self.lyr.append(nn.Linear(25,3)); self.lyr.append(nn.LogSoftmax(dim=1))
        
    def forward(self, x):
        y = x
        for l in self.lyr:
            y = l(y)
        return y

def learn(net, x, t, lrate=0.1, epochs=10):
    '''
    Perform learning by backprop.
    This method also calculates the magnitude of the error gradients
    for the deepest layer of connection weights.
    Output:
     bottom_grad  a list of the average magnitude of the
                  gradients at the deepest layer
     losses       a list containing the losses over training epochs
    '''
    optim = torch.optim.SGD(net.parameters(), lr=lrate)
    bottom_grad = []
    losses = []
    for epoch in range(epochs):
        y = net(x)
        loss = loss_func(y, t)
        optim.zero_grad()
        loss.backward()
        losses.append(loss.item())
        bottom_grad.append(torch.mean(torch.abs(list(net.parameters())[0].grad)))
        optim.step()
    return bottom_grad, losses
