def Wvec(N,k):
    W = np.exp(2.j*np.pi/N)
    Wk = []
    for n in range(N):
        Wk.append(W**(n*k))
    return np.array(Wk)

DFT Matrix¶

def DFTMatrix(N):
    M = []
    for k in range(N):
        M.append(np.conj(Wvec(N,k)))
    return np.array(M)
