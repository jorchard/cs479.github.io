def EvalPoly(x, coefs):
    '''
     y = EvalPoly(x, coefs)
     
     Evaluates the linear model
     
       y = coefs[0] + coefs[1]*x + coefs[2]*x**2 + ... + coefs[n]*x**n
    '''
    y = np.zeros(np.shape(x))
    for idx,c in enumerate(coefs):
        y += c*x**idx
    return y
